    <script>
        // --- Mock de Datos ---
        const productData = {
            thinkpad: {
                brand: 'Lenovo',
                model: 'ThinkPad T14 Gen 2',
                ram: {
                    support: true,
                    totalSlots: 2,
                    occupiedSlots: 1,
                    maxGB: 64,
                    type: 'DDR4 SODIMM'
                },
                storage: {
                    support: true,
                    maxTotal: '2 TB',
                    type: 'NVMe M.2 2280'
                }
            },
            macbook: {
                brand: 'Apple',
                model: 'MacBook Air M1',
                ram: {
                    support: false,
                    totalSlots: 0,
                    occupiedSlots: 0,
                    maxGB: 16,
                    type: 'LPDDR4X (Soldada)'
                },
                storage: {
                    support: false,
                    maxTotal: '2 TB',
                    type: 'SSD Integrado (Soldado)'
                }
            }
        };

        // --- Referencias a Elementos del DOM ---
        const ui = {
            brand: document.getElementById('product-brand'),
            model: document.getElementById('product-model'),
            ram: {
                statusIndicator: document.getElementById('ram-status-indicator'),
                statusIcon: document.getElementById('ram-status-icon'),
                statusText: document.getElementById('ram-status-text'),
                slotsContainer: document.getElementById('ram-slots-container'),
                slotsVisual: document.getElementById('ram-slots-visual'),
                gaugeContainer: document.getElementById('ram-gauge-container'),
                maxValue: document.getElementById('ram-max-value'),
                progressBar: document.getElementById('ram-progress-bar'),
                infoText: document.getElementById('ram-info-text')
            },
            storage: {
                statusIndicator: document.getElementById('storage-status-indicator'),
                statusIcon: document.getElementById('storage-status-icon'),
                statusText: document.getElementById('storage-status-text'),
                detailsContainer: document.getElementById('storage-details-container'),
                typeText: document.getElementById('storage-type-text'),
                gaugeContainer: document.getElementById('storage-gauge-container'),
                maxValue: document.getElementById('storage-max-value'),
                progressBar: document.getElementById('storage-progress-bar'),
                infoText: document.getElementById('storage-info-text')
            }
        };
        
        // --- Iconos SVG ---
        const icons = {
            check: '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path>',
            cross: '<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path>'
        };

        // --- Función para animar números ---
        function animateValue(element, start, end, duration) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                element.innerText = Math.floor(progress * (end - start) + start);
                if (progress < 1) {
                    window.requestAnimationFrame(step);
                }
            };
            window.requestAnimationFrame(step);
        }

        // --- Función principal para actualizar el Dashboard ---
        function updateDashboard(product) {
            // Actualizar encabezado
            ui.brand.innerText = product.brand;
            ui.model.innerText = product.model;

            // --- Lógica para RAM ---
            if (product.ram.support) {
                // Estilos para 'Ampliable'
                ui.ram.statusIndicator.className = 'w-16 h-16 rounded-full flex items-center justify-center bg-green-500/10 text-green-400 animate-pulse-green';
                ui.ram.statusIcon.innerHTML = icons.check;
                ui.ram.statusText.innerText = 'Ampliable';
                ui.ram.statusText.className = 'text-2xl font-bold text-green-400';
                ui.ram.progressBar.className = 'progress-bar-inner h-4 rounded-full bg-gradient-to-r from-green-500 to-cyan-500';
                ui.ram.infoText.innerText = `Este equipo tiene slots disponibles para expansión.`;
                
                // Mostrar elementos visuales
                ui.ram.slotsContainer.style.display = 'block';
                ui.ram.gaugeContainer.style.display = 'block';
                
                // Dibujar slots
                ui.ram.slotsVisual.innerHTML = '';
                for (let i = 0; i < product.ram.totalSlots; i++) {
                    const slot = document.createElement('div');
                    slot.className = `w-full rounded ${i < product.ram.occupiedSlots ? 'bg-cyan-500' : 'bg-slate-700/50 border-2 border-dashed border-slate-600'}`;
                    ui.ram.slotsVisual.appendChild(slot);
                }
            } else {
                // Estilos para 'No Ampliable'
                ui.ram.statusIndicator.className = 'w-16 h-16 rounded-full flex items-center justify-center bg-red-500/10 text-red-400';
                ui.ram.statusIcon.innerHTML = icons.cross;
                ui.ram.statusText.innerText = 'No Ampliable';
                ui.ram.statusText.className = 'text-2xl font-bold text-red-400';
                ui.ram.progressBar.className = 'progress-bar-inner h-4 rounded-full bg-slate-600';
                ui.ram.infoText.innerText = `La memoria viene soldada en placa y no puede ser modificada.`;

                // Ocultar elementos visuales
                ui.ram.slotsContainer.style.display = 'none';
                ui.ram.gaugeContainer.style.display = 'none';
            }
            // Animar valor de RAM máxima y barra de progreso
            animateValue(ui.ram.maxValue, 0, product.ram.maxGB, 800);
            setTimeout(() => {
                ui.ram.progressBar.style.width = product.ram.support ? `100%` : `0%`;
            }, 100);


            // --- Lógica para Almacenamiento ---
            if (product.storage.support) {
                // Estilos para 'Ampliable'
                ui.storage.statusIndicator.className = 'w-16 h-16 rounded-full flex items-center justify-center bg-green-500/10 text-green-400 animate-pulse-green';
                ui.storage.statusIcon.innerHTML = icons.check;
                ui.storage.statusText.innerText = 'Ampliable';
                ui.storage.statusText.className = 'text-2xl font-bold text-green-400';
                ui.storage.progressBar.className = 'progress-bar-inner h-4 rounded-full bg-gradient-to-r from-green-500 to-cyan-500';
                ui.storage.infoText.innerText = `Es posible cambiar o añadir unidades de almacenamiento.`;

                // Mostrar elementos
                ui.storage.detailsContainer.style.display = 'block';
                ui.storage.gaugeContainer.style.display = 'block';
                ui.storage.typeText.innerText = product.storage.type;
            } else {
                // Estilos para 'No Ampliable'
                ui.storage.statusIndicator.className = 'w-16 h-16 rounded-full flex items-center justify-center bg-red-500/10 text-red-400';
                ui.storage.statusIcon.innerHTML = icons.cross;
                ui.storage.statusText.innerText = 'No Ampliable';
                ui.storage.statusText.className = 'text-2xl font-bold text-red-400';
                ui.storage.progressBar.className = 'progress-bar-inner h-4 rounded-full bg-slate-600';
                ui.storage.infoText.innerText = `El almacenamiento está integrado y no puede ser modificado.`;
                
                // Ocultar elementos
                ui.storage.detailsContainer.style.display = 'none';
                ui.storage.gaugeContainer.style.display = 'none';
            }
            // Actualizar valor de almacenamiento
            ui.storage.maxValue.innerText = product.storage.maxTotal;
             setTimeout(() => {
                ui.storage.progressBar.style.width = product.storage.support ? `100%` : `0%`;
            }, 100);
        }

        // --- Event Listeners para los botones ---
        const btnThinkpad = document.getElementById('btn-thinkpad');
        const btnMacbook = document.getElementById('btn-macbook');

        btnThinkpad.addEventListener('click', () => {
            updateDashboard(productData.thinkpad);
            btnThinkpad.classList.remove('bg-slate-700', 'hover:bg-slate-600');
            btnThinkpad.classList.add('bg-blue-600', 'hover:bg-blue-500');
            btnMacbook.classList.add('bg-slate-700', 'hover:bg-slate-600');
            btnMacbook.classList.remove('bg-blue-600', 'hover:bg-blue-500');
        });

        btnMacbook.addEventListener('click', () => {
            updateDashboard(productData.macbook);
            btnMacbook.classList.remove('bg-slate-700', 'hover:bg-slate-600');
            btnMacbook.classList.add('bg-blue-600', 'hover:bg-blue-500');
            btnThinkpad.classList.add('bg-slate-700', 'hover:bg-slate-600');
            btnThinkpad.classList.remove('bg-blue-600', 'hover:bg-blue-500');
        });

        // --- Carga inicial ---
        window.addEventListener('load', () => {
            updateDashboard(productData.thinkpad);
        });

    </script>
